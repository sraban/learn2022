Q. When States are inter-dependent - don't modify in same function ... updating each state takes sometime to render the component.

   Ex. 
   	function Component({}) {
      const calculation = () => {
      	setAddress()
      	setContact()
      }
    }

   Sol. Works one function after another synchronously -- states are updated one after other with tpl render each
    function Component({}) {
      const calculation = () => {
      	setAddress()
      	contact();
      }
      
      const contact = () => {
      	setContact();
      }
    }
